        'pid': pid,
        'msg': msg,
        'event_type': None,
        'server_ip': None,
        'xid': None,
    }

    mo = OFFER_RE.search(msg)
    if mo:
        rec['event_type'] = 'offer'
        rec['server_ip'] = mo.group(1)
        mx = XID_ANY_RE.search(msg)

        rec['event_type'] = 'ack'
        rec['server_ip'] = ma.group(1)
        rec['xid'] = ma.group(2)
        return rec

    if mr:
        rec['event_type'] = 'request'
        rec['server_ip'] = mr.group(2)
        rec['xid'] = mr.group(3)
        return rec

def cleanup_pending(pending, ts_now):
    limit = ts_now - 60
    while pending and pending[0]['ts'] < limit:
        pending.popleft()


def match_ack(pending, ack_ts, host, pid, xid):
    cleanup_pending(pending, ack_ts)
    idx = None
    if xid is not None:
        for i, off in enumerate(pending):
            if off['host'] == host and off['xid'] == xid and 0 <= ack_ts - off['ts'] <= 60:
                idx = i
                break
    else:
        candidates = [i for i, off in enumerate(pending) if off['host'] == host and off['pid'] == pid and off['pid'] is not None and 0 <= ack_ts - off['ts'] <= 60]
        if len(candidates) == 1:
            idx = candidates[0]
    if idx is None:
        return False
    del pending[idx]
    return True

def split_embedded(line):
    parts = []
    s = line.rstrip('\n')
    while s:
        m = embedded_re.search(s)
        if not m:
            parts.append(s)
            break
        parts.append(s[:m.start()].rstrip())
        embedded = m.group(1)
        s = embedded
    return [p for p in parts if p]
