import re

# Match standard syslog prefix up to dhclient[pid]: or dhclient:
prefix_re = re.compile(r'^[A-Z][a-z]{2}\s+\d{1,2}\s+\d{2}:\d{2}:\d{2}\s+\S+\s+dhclient(?:\[(\d+)\])?:\s*(.*)$')

# Primary DHCP message patterns that include interface directly.
direct_patterns = [
    re.compile(r'^(DHCPDISCOVER|DHCPREQUEST|DHCPRELEASE|DHCPDECLINE)\s+on\s+(\S+)\b'),
    re.compile(r'^XMT:\s*([^\s].*?)\s+on\s+(\S+),\s+interval\b'),
    re.compile(r'^RCV:\s*([^\s].*?)\s+message\s+on\s+(\S+)\s+from\s+'),
]

# Patterns that can provide interface context for later DHCP message lines lacking interface.
context_patterns = [
    re.compile(r'^(?:Listening|Sending)\s+on\s+LPF/(\S+?)(?:/\S*)?\s*$'),
    re.compile(r'^(?:Listening|Sending)\s+on\s+Socket/(\S+)\s*$'),
    re.compile(r'^receive_packet\s+failed\s+on\s+(\S+):'),
    re.compile(r'^PRC:\s+[^.]*\s+lease\s+on\s+(\S+)\.'),
    re.compile(r'^no\s+link-local\s+IPv6\s+address\s+for\s+(\S+)\b'),
]

# DHCP message lines without interface; emit only if interface can be inferred from context.
message_only_patterns = [
    re.compile(r'^(DHCPACK)\s+from\b'),
    re.compile(r'^(DHCPOFFER)\s+from\b'),
    re.compile(r'^(DHCPNAK)\s+from\b'),
]

# Handle concatenated syslog payloads such as: ... <30>Dec 28 19:30:22 dhclient: DHCPREQUEST on eth0 ...
embedded_re = re.compile(r'<\d+>[A-Z][a-z]{2}\s+\d{1,2}\s+\d{2}:\d{2}:\d{2}\s+dhclient(?:\[(\d+)\])?:\s*')


def process_message(msg, host, pid, ctx_by_host_pid, ctx_by_host):
    out = []

    # First, split possible embedded records while keeping order.
    parts = []
    last = 0
    for m in embedded_re.finditer(msg):
        if m.start() > last:
            parts.append((pid, msg[last:m.start()].strip()))
        last = m.end()
        next_m = embedded_re.search(msg, last)
        if next_m:
            parts.append((m.group(1), msg[last:next_m.start()].strip()))
            last = next_m.start()
    if last == 0:
        parts = [(pid, msg.strip())]
    elif last < len(msg):
        tail_match = None
        for m in embedded_re.finditer(msg):
            tail_match = m
        parts.append((tail_match.group(1) if tail_match else pid, msg[last:].strip()))

    for cur_pid, part in parts:
        if not part:
            continue

        matched = False
        for pat in direct_patterns:
            m = pat.search(part)
            if m:
                message_type = m.group(1).strip()
                interface = m.group(2).strip()
                if interface and message_type:
                    out.append((interface, message_type))
                    if cur_pid is not None:
                        ctx_by_host_pid[(host, cur_pid)] = interface
                    ctx_by_host[host] = interface
                matched = True
                break
        if matched:
            continue

        for pat in context_patterns:
            m = pat.search(part)
            if m:
                interface = m.group(1).strip()
                if interface:
                    if cur_pid is not None:
                        ctx_by_host_pid[(host, cur_pid)] = interface
                    ctx_by_host[host] = interface
                matched = True
                break
        if matched:
            continue

        for pat in message_only_patterns:
            m = pat.search(part)
            if m:
                message_type = m.group(1).strip()
                interface = None
                if cur_pid is not None:
                    interface = ctx_by_host_pid.get((host, cur_pid))
                if not interface:
                    interface = ctx_by_host.get(host)
                if interface:
                    out.append((interface, message_type))
                break

    return out


def main():
    if len(sys.argv) < 2:
        return

    path = sys.argv[1]
    ctx_by_host_pid = {}
    ctx_by_host = {}

    with open(path, 'r', encoding='utf-8', errors='replace') as f:
        for line in f:
            line = line.rstrip('\n')
            m = prefix_re.match(line)
            if not m:
                continue
            pid, msg = m.groups()

            # Extract host conservatively from prefix without brittle full parsing.
            tokens = line.split()
            if len(tokens) < 5:
                continue
            host = tokens[3]

            for interface, message_type in process_message(msg, host, pid, ctx_by_host_pid, ctx_by_host):
                sys.stdout.write(f"{interface} {message_type}\n")
