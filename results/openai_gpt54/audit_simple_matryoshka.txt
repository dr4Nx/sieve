def extract_type(line):

    return m.group(1) if m else None

    return m.group(1) if m else ''

def extract_exe(text):
    m = EXE_RE.search(text)
    if not m:
        return None

    if exe in SSHD_EXES:
        return True
    exe2 = extract_exe(line)
    if exe2 in SSHD_EXES:
        return True
    term = TERM_RE.search(inner)
    if term and term.group(1) in SSHD_TERMINALS:
        return True
    return False

    if 'sshd' not in line:
        return False

    event_type = extract_type(line)
    inner = extract_inner_msg(line)

    if not is_failed(inner):
        return False

        return False

            return 'authentication' in op.lower()

    if PAM_STYLE_RE.search(inner):
        return True

    if 'acct=' in inner and 'sshd' in line:
        return True

    return False
