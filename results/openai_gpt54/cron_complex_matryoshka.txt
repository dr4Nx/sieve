            dt = dt.replace(microsecond=us)
        if tzs and tzs != 'Z':

            hh = int(tzs[1:3])
            mm = int(tzs[4:6])
            offset = timezone(sign * timedelta(hours=hh, minutes=mm))
            dt = dt.replace(tzinfo=offset)
            return dt.timestamp()
        if tzs == 'Z':
            dt = dt.replace(tzinfo=timezone.utc)
            return dt.timestamp()
