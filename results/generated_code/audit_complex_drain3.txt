    counts = defaultdict(int)

    def is_valid(val):
        return val.lower() not in ('none', 'null', 'n/a', '(null)', '?')

    try:
        with opener(log_file, 'rt', encoding='utf-8', errors='ignore') as f:
            for line in f:
                # Handle syslog messages containing multiple '#012' separated audit records
                segments = line.split('#012')
                for segment in segments:
                    type_m = type_re.search(segment)
                    if not type_m:
                        continue
                    
                    t_type = type_m.group(1)
                    
                    audit_m = audit_id_re.search(segment)
                    if not audit_m:
                        # Inline fallback for records missing a transaction ID
                        if t_type in ('1701', 'ANOM_ABEND'):
                            exe_m = exe_re.search(segment)
                            sig_m = sig_re.search(segment)
                            if exe_m and sig_m:
                                exe = exe_m.group(1) or exe_m.group(2)
                                sig = sig_m.group(1)
                                if is_valid(exe) and is_valid(sig):
                                    counts[(exe, sig)] += 1
                        continue

                    node_m = node_re.search(segment)
                    node = node_m.group(1) if node_m else ''
                    
                    # Key by node and transaction_id to prevent false global merges
                    tx_key = (node, audit_id)
