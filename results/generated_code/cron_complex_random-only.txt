def get_ts_val(ts_text):

    if '.' in ts_str:
        fmt += '.%f'
    if '+' in ts_str or ('-' in ts_str[10:] and 'T' in ts_str):
        fmt += '%z'
    
    try:
        return datetime.strptime(ts_str, fmt).timestamp()
    except Exception:
        try:
            return datetime.fromisoformat(ts_text).timestamp()
        except Exception:
            try:
                # Fallback for standard syslog timestamps
                ts_clean = re.sub(r'\s+', ' ', ts_text)
                return datetime.strptime(ts_clean, '%b %d %H:%M:%S').replace(year=2000).timestamp()
            except Exception:
                return ts_text
