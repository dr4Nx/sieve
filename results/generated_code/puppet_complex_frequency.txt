        r'(?:[^:\[]+)(?:\[(\d+)\])?:\s+'                # Group 3: process_id (optional)

    skip_events = set()

    with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            # Fast pre-check to speed up parsing
            if 'failed dependencies' in line or 'has failures: true' in line:

                    process_id = match.group(3) or 'unknown_pid'

                        event_key = (host, process_id, resource_id)

def process_puppet_logs(file_path):
    # Regex to match standard syslog formats from puppet or puppet-agent with PID
    log_pattern = re.compile(
        r'^([A-Z][a-z]{2}\s+\d+\s+\d{2}:\d{2}:\d{2})\s+'  # Syslog timestamp (Group 1)
        r'(\S+)\s+'                                      # Hostname (Group 2)
        r'(?:[^\s:\[]*puppet[^\s:\[]*)\s*\[(\d+)\]:\s+'  # Process containing 'puppet' and its PID (Group 3)
        r'(.*)$'                                          # Message (Group 4)
    )

    # Tracks sequence state for each (host, pid) pair
    states = {}

    with open(file_path, 'r', encoding='utf-8', errors='replace') as f:
        for line in f:
            match = log_pattern.match(line)
            if not match:
                continue
            
            timestamp, host, pid, message = match.groups()
            key = (host, pid)

            # Clear state on indications of a new puppet run to prevent cross-run false positives
            if message.startswith("Loading facts") or \
               message.startswith("Starting Puppet client") or \
               message.startswith("Retrieving plugin"):
                if key in states:
                    del states[key]

            # Event 1: Remote catalog retrieval failure
            if "Could not retrieve catalog from remote server" in message:
                states[key] = {'state': 1, 't1': timestamp}
            
            # Event 2: Use of a cached catalog
            elif "Using cached catalog" in message:
                if states.get(key, {}).get('state') == 1:
                    states[key]['state'] = 2
                    states[key]['t2'] = timestamp
            
            # Event 3: Report-send failure
            elif "Could not send report" in message:
                if states.get(key, {}).get('state') == 2:
                    t1 = states[key]['t1']
                    t2 = states[key]['t2']
                    t3 = timestamp
                    # Output matches requested format: host process_id catalog_failure_timestamp cached_catalog_timestamp report_failure_timestamp
                    print(f"{host} {pid} {t1} {t2} {t3}")
                    # Reset state after emitting a full match
                    del states[key]
