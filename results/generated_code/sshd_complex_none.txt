    if re.match(r'^(?:\d{1,3}\.){3}\d{1,3}$', s):
        return True
    if ':' in s and (s.count(':') >= 2 or '::' in s) and re.match(r'^[a-fA-F0-9:]+$', s):
        return True
    return False
