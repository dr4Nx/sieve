    # Support various timestamp formats: standard syslog, ISO 8601, and RFC 3339
    ts_regex = r'(?:(?:[0-9]{4}\s+)?[A-Z][a-z]{2}\s+\d+\s+\d{2}:\d{2}:\d{2}|[0-9]{4}-[0-9]{2}-[0-9]{2}T[0-9:.Z+-]+|[0-9]{4}-[0-9]{2}-[0-9]{2}\s+[0-9]{2}:\d{2}:\d{2}(?:\.\d+)?)'
    syslog_re = re.compile(rf'^(?:<[0-9]+>)?{ts_regex}\s+(\S+)\s+sshd[^:]*:\s+(.*)')

                host = m.group(1)

                # Check for successful authentication
                m_acc = accept_re.search(msg)
