    renewal_re = re.compile(r'bound to \S+ -- renewal in (\d+) seconds')
    
    # Matches explicit lease durations in other formats like "lease 86401", "duration 86401"
    explicit_lease_re = re.compile(r'(?:lease|duration|valid for)(?:\s+of)?\s+(\d+)\s*(?:seconds?)?', re.IGNORECASE)
    
    with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            renewal_match = renewal_re.search(line)
            if renewal_match:
                renewal_time = int(renewal_match.group(1))
                if renewal_time > 43200:
                    print(line, end='')

            # Check if it's an IP assignment line and explicitly states a duration
            if re.search(r'(DHCPACK|bound|assigned|lease)', line, re.IGNORECASE):
                duration_match = explicit_lease_re.search(line)
                if duration_match:
                    duration_time = int(duration_match.group(1))
                    if duration_time > 86400:
                        print(line, end='')

def parse_syslog_dhcp(line):
    # Parse standard syslog prefix
    match = re.match(
        r'^(?P<month>[A-Z][a-z]{2})\s+(?P<day>\d+)\s+(?P<time>\d{2}:\d{2}:\d{2})\s+(?P<host>\S+)\s+(?P<proc_pid>[^:]+):\s+(?P<msg>.*)',
        line
    )
    if not match:
        return None
        
    record = match.groupdict()
    msg = record['msg']
    
    # Build explicit intermediate records as requested
    record['event_type'] = None
    record['interface'] = None
    record['port'] = None
    
    event_match = re.search(r'\b(DHCPREQUEST|DHCPACK|DHCPDISCOVER|DHCPDECLINE|DHCPRELEASE|DHCPINFORM)\b', msg)
    if event_match:
        record['event_type'] = event_match.group(1)
        
    iface_match = re.search(r'\bon\s+([a-zA-Z0-9_-]+)\b', msg)
    if iface_match:
        record['interface'] = iface_match.group(1)
        
    port_match = re.search(r'\bport\s+(\d+)\b', msg)
    if port_match:
        record['port'] = port_match.group(1)
