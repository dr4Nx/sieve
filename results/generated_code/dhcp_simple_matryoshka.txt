def process_logs(filepath):
    # Regex to match syslog header and extract host
    host_regex = re.compile(r'(?:^|<\d+>)[A-Z][a-z]{2}\s+\d+\s+\d{2}:\d{2}:\d{2}\s+(?P<host>\S+)')
    # Regex to isolate the dhclient message part, capturing optional PID
    dhclient_regex = re.compile(r'\bdhclient(?:\[(?P<pid>\d+)\])?:\s+(?P<msg>.*)')
    # Regex for state tracking of IP being attempted
    trying_lease_regex = re.compile(r'Trying recorded lease\s+(?P<ip>[0-9a-fA-F\.:]+)')
    # Regex for bound message to capture IP and duration
    bound_regex = re.compile(r'bound(?:(?:\s+to\s+(?P<ip>[0-9a-fA-F\.:]+))|\s+to\s+--|:)?\s*(?:--\s*)?renewal in\s+(?P<duration>\d+)')
    
    ips = set()
    # State map: (host, pid) -> IP
    pending_leases = {}
    
    open_func = gzip.open if filepath.endswith('.gz') else open
    
    try:
        with open_func(filepath, 'rt', encoding='utf-8', errors='ignore') as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                    
                host_match = host_regex.search(line)
                host = host_match.group('host') if host_match else "unknown"
                
                # Handle edge cases where multiple messages are packed into a single line (e.g. syslog repetition)
                parts = re.split(r'<\d+>', line)
                for part in parts:
                    m = dhclient_regex.search(part)
                    if m:
                        pid = m.group('pid')
                        msg = m.group('msg').strip()
                        
                        t_match = trying_lease_regex.search(msg)
                        if t_match:
                            ip = t_match.group('ip').rstrip('.')
                            pending_leases[(host, pid)] = ip
                            continue
                            
                        b_match = bound_regex.search(msg)
                        if b_match:
                            ip = b_match.group('ip')
                            if ip:
                                ip = ip.rstrip('.')
                            else:
                                # Fallback to state if IP is omitted in the bound log message
                                ip = pending_leases.get((host, pid))
                                
                            duration = int(b_match.group('duration'))
                            
                            # If the renewal duration is over 1 day (86400 seconds)
                            if ip and duration > 86400:
                                ips.add(ip)
    except Exception:
        pass
        
    for ip in sorted(ips):
        print(ip)
