def is_auth_failure(msg):
    msg_lower = msg.lower()
    if "authentication failure" in msg_lower:
        return True
    if "invalid user" in msg_lower:
        return True
    if msg.startswith("Failed "):
        return True
    if "bad username" in msg_lower:
        return True
    if "user unknown" in msg_lower:
        return True
    return False

def extract_ip(msg):
    m = re.search(r"rhost=([^\s]+)", msg)
    if m:
        return m.group(1).strip(":")
    
    m = re.search(r"(?:for \S+|user \S+|string|disconnect)\s+from\s+([^\s]+)", msg)
    if m:
        ip = m.group(1).strip(":")
        if not ip.startswith("system_u") and not ip.startswith("unconfined_u"):
            return ip
            
    m = re.search(r"Connection closed by\s+([^\s]+)", msg)
    if m and m.group(1) not in ("remote", "invalid", "UNIX", "authenticating"):
        return m.group(1).strip()
        
    m = re.search(r"getaddrinfo for \. \[([^\]]+)\]", msg)
    if m:
        return m.group(1).strip()

    m = re.search(r"\bfrom\s+([^\s]+)", msg)
    if m:
        ip = m.group(1).strip(":")
        if not ip.startswith("system_u") and not ip.startswith("unconfined_u"):
            return ip

    return None
