def process_audit_logs(file_path):

        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            for raw_line in f:
                syslog_host = get_syslog_host(raw_line)
                
                # Handle lines packed by setroubleshoot with #012 (newline placeholder)
                for line in raw_line.split('#012'):
                    tx_match = tx_re.search(line)
                    if tx_match:
                        tx_id = tx_match.group('tx')
                    else:
                        # Fallback to PID if transaction ID is missing entirely
                        pid_match = pid_re.search(line)
                        if pid_match:
                            tx_id = f"pid_{pid_match.group(1)}"
                        else:
                            continue
                            
                    node_match = node_re.search(line)
                    client = ""
                    if node_match:
                        client = node_match.group(1)
                    elif syslog_host:
                        client = syslog_host
                        
                    key = (client, tx_id)
                    
                    # Extract executable path
                    em = exe_re.search(line)
                    if em:
                        exe_val = em.group(1) if em.group(1) is not None else em.group(2)
                        if exe_val not in ('?', '(null)', '""', "''"):
                            tx_store[key]['exe'] = exe_val
                            
                    # Check for crash signal events (ANOM_ABEND or 1701)
                    tm = type_re.search(line)
                    if tm:
                        etype = tm.group(1)
                        if etype in ('1701', 'ANOM_ABEND'):
                            sm = sig_re.search(line)
                            if sm:
                                tx_store[key]['sigs'].add(sm.group(1))
                                
    except Exception:
        pass
        
    counts = defaultdict(int)
    for key, data in tx_store.items():
        exe = data['exe']
        if not exe:
            continue
        for sig in data['sigs']:
            counts[(exe, sig)] += 1
            
    for (exe, sig) in sorted(counts.keys()):
