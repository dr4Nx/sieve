def extract_host(line):
    # Priority 1: 'node=' attribute, typically used by audispd/setroubleshoot for explicit host
    m = re.search(r'\bnode=([^\s]+)', line)
    if m:
        return m.group(1)

    # Matches `<Host> <Service>:` while avoiding matching the seconds of a HH:MM:SS timestamp
    m = re.search(r'(?<!:)\b([A-Za-z0-9_.-]+)\s+(?:kernel|audispd|audit_log|klogd|setroubleshoot)(?:\[\d+\])?:\s', line)
    if m:
        host = m.group(1)
        # Skip if it accidentally matched AM/PM from a 12-hour timestamp format
        if host not in ('AM', 'PM'):
            return host
            
    return 'unknown'
