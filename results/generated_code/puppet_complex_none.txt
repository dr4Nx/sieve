def parse_puppet_logs(file_path):

        r'^\s*(?:<\d+>)?'

    )
    
    scheduled_pattern = re.compile(r"\bschedul(?:ed|ing)\s+['\"]?refresh['\"]?", re.IGNORECASE)
    triggered_pattern = re.compile(r"\btrigger(?:ed|ing)\s+['\"]?refresh['\"]?", re.IGNORECASE)
    
    open_func = gzip.open if file_path.endswith('.gz') else open
    
    try:
        with open_func(file_path, 'rt', encoding='utf-8', errors='replace') as f:
            for line in f:
                match = log_pattern.search(line)
                if match:
                    host = match.group(1)
                    pid = int(match.group(2))
                    msg = match.group(3)

                        stats[(host, pid)]['triggered'] += int(is_triggered)
    except Exception:
