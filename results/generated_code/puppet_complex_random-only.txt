        r'(puppet[^\s\[:]*)\[(\d+)\]:\s*(.*)'
    )
    
    sched_pattern = re.compile(r'Schedul(?:ing|ed)\s+refresh', re.IGNORECASE)
    trig_pattern = re.compile(r"Trigger(?:ed|ing)\s+'refresh'", re.IGNORECASE)
    
    records = []
    
    opener = gzip.open if file_path.endswith('.gz') else open
    try:
        with opener(file_path, 'rt', encoding='utf-8', errors='replace') as f:
            for line in f:

                if not m:
                    continue
                
                host = m.group(1)
                pid_str = m.group(3)
                msg = m.group(4)
                
                pid = int(pid_str)
                
                scheduled = sched_pattern.search(msg) is not None
                triggered = trig_pattern.search(msg) is not None
                
                # Exclude dry-run "Would have" events to strictly count actual scheduled/triggered refreshes
                if scheduled and re.search(r'have\s+Schedul', msg, re.IGNORECASE):
                    scheduled = False
                if triggered and re.search(r'have\s+Trigger', msg, re.IGNORECASE):
                    triggered = False

                    records.append({
                        'host': host,
                        'process_id': pid,

                        'triggered': 1 if triggered else 0
                    })
    except Exception:
        pass
        
    # Aggregate explicit intermediate records by host and process ID
    state = defaultdict(lambda: [0, 0])
    for r in records:
        key = (r['host'], r['process_id'])
        state[key][0] += r['scheduled']
        state[key][1] += r['triggered']
        
    # Output fields: host, process_id, scheduled_refresh_count, triggered_refresh_count
    for (host, pid), counts in state.items():
