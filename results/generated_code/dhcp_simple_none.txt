def parse_seconds(val, unit):
    val = int(val)
    if not unit:
        return val
    u = unit.lower()
    if u.startswith('d'): return val * 86400
    if u.startswith('h'): return val * 3600
    if u.startswith('m'): return val * 60
    return val
