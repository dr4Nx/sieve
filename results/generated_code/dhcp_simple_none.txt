def to_seconds(val, unit):
    val = float(val)
    if not unit:
        return val
    unit = unit.lower()

    if unit.startswith('m'): return val * 60
    if unit.startswith('h'): return val * 3600
    if unit.startswith('d'): return val * 86400
    return val

def is_interface(s):
    if not s:
        return False
    s_lower = s.lower()
    if s_lower in ('port', 'interval', 'ip', 'address', 'xid'):
        return False
    if re.match(r'^\d{1,3}(\.\d{1,3}){3}$', s):
        return False
    if s.count(':') >= 2:
        return False
    if re.match(r'^([0-9a-fA-F]{2}-){5}[0-9a-fA-F]{2}$', s):
        return False

def process_logs(file_path):
    msg_type_re = re.compile(r'\b(DHCP[A-Z]+)\b')
    dnsmasq_re = re.compile(r'\b(DHCP[A-Z]+)\(([a-zA-Z0-9_.:-]+)\)')
    via_re = re.compile(r'\bvia\s+([a-zA-Z0-9_.:-]+)')
    on_re = re.compile(r'\bon\s+([a-zA-Z0-9_.:-]+)')

    with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            m_dnsmasq = dnsmasq_re.search(line)
            if m_dnsmasq:
                msg_type = m_dnsmasq.group(1)
                interface = m_dnsmasq.group(2)
                if is_interface(interface):
                    print(f"{interface} {msg_type}")
                continue

            m_type = msg_type_re.search(line)
            if m_type:
                msg_type = m_type.group(1)
                interface = None

                for candidate in via_re.findall(line):
                    if is_interface(candidate):
                        interface = candidate
                        break
                
                if not interface:
                    for candidate in on_re.findall(line):
                        if is_interface(candidate):
                            interface = candidate
                            break
