        for line in f:
            m = log_pattern.match(line)
            if not m:
                continue
            
            _, host, process_ident, msg = m.groups()
            
            if not process_ident.startswith('puppet'):
                continue
                
            pid_match = re.search(r'\[(\d+)\]', process_ident)
            pid = pid_match.group(1) if pid_match else 'no_pid'
            
            if msg.startswith("Applying configuration version"):
                started[host] = started.get(host, 0) + 1
                active_runs[(host, pid)] = True
            elif msg.startswith("Finished catalog run") or msg.startswith("Applied catalog"):
                if active_runs.get((host, pid)):
                    finished[host] = finished.get(host, 0) + 1
                    active_runs[(host, pid)] = False
                    
    for host in sorted(started.keys()):
        s_count = started[host]
        f_count = finished.get(host, 0)
        rate = float(f_count) / s_count if s_count > 0 else 0.0
        print(f"{host} {s_count} {f_count} {rate}")
