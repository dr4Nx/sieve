def parse_cron_logs(file_path):
    # Pattern to match the command executed by cron, e.g., CMD ( <command> )
    cmd_pattern = re.compile(r'\bCMD\s+\((.*)\)\s*$')
    distinct_commands = set()

    # Support both plain text and gzipped logs
    open_func = gzip.open if file_path.endswith('.gz') else open

    try:
        with open_func(file_path, 'rt', encoding='utf-8', errors='ignore') as f:
            for line in f:
                match = cmd_pattern.search(line)
                if match:
                    # Extract the command, stripping any leading/trailing whitespace
                    command = match.group(1).strip()
                    if command:
                        distinct_commands.add(command)
    except Exception:
        pass

    # Print distinct commands, one per line
    for cmd in sorted(distinct_commands):
        print(cmd)
