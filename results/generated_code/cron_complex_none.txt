def parse_time(line, current_year, last_month):

            if len(tz) == 3:
                tz += '00'
        else:
            tz = ''
        
        ts_str = base + tz
        try:
            if '.' in base:
                if tz:
                    return datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%S.%f%z").timestamp(), current_year, last_month
                else:
                    return datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%S.%f").timestamp(), current_year, last_month

                    return datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%S%z").timestamp(), current_year, last_month

        r'^(?P<date>\d{4}-\d{2}-\d{2})[T\s](?P<hour>\d{2}):(?P<minute>\d{2})'
        r'|'
        r'^(?P<syslog_date>[A-Z][a-z]{2}\s+\d+)\s+(?P<syslog_hour>\d{2}):(?P<syslog_minute>\d{2})'
    )

        for line in f:
            # Fast pre-checks before running regexes
            if 'CMD' not in line:
                continue
            if 'cron' not in line.lower():
                continue
            if not cmd_regex.search(line):
                continue

            m = dt_regex.search(line)
            if not m:
                continue

            d = m.groupdict()
            if d['date']:
                date_str = d['date']
                hour = int(d['hour'])

                minute = int(d['syslog_minute'])
