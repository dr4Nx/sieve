class LogRecord:
    def __init__(self, raw_line):
        self.raw_line = raw_line
        self.timestamp = None
        self.host = None
        self.process = None
        self.process_id = None
        self.message = None
        
        self.event_type = None
        self.transaction_id = None
        self.interface = None
        self.ip = None
        
        self._parse()
        
    def _parse(self):
        m = re.match(r'^([A-Z][a-z]{2}\s+\d+\s+\d{2}:\d{2}:\d{2})\s+(\S+)\s+([^:]+):\s+(.*)', self.raw_line)
        if m:
            self.timestamp = m.group(1)
            self.host = m.group(2)
            proc_full = m.group(3)
            self.message = m.group(4)
            
            proc_m = re.match(r'([a-zA-Z0-9_-]+)(?:\[(\d+)\])?', proc_full)
            if proc_m:
                self.process = proc_m.group(1)
                self.process_id = proc_m.group(2)
            else:
                self.process = proc_full
                
            if self.message:
                xid_m = re.search(r'\(xid=(0x[0-9a-fA-F]+)\)', self.message)
                if xid_m:
                    self.transaction_id = xid_m.group(1)
                    
                iface_m = re.search(r'(DHCP[A-Z]+)\s+on\s+([a-zA-Z0-9_:-]+)', self.message)
                if iface_m:
                    self.event_type = iface_m.group(1)
                    self.interface = iface_m.group(2)
                else:
                    lpf_m = re.search(r'(Listening|Sending)\s+on\s+(?:LPF|Socket)/([a-zA-Z0-9_:-]+)', self.message)
                    if lpf_m:
                        self.event_type = lpf_m.group(1)
                        self.interface = lpf_m.group(2)
                    else:
                        fail_m = re.search(r'(receive_packet failed) on ([a-zA-Z0-9_:-]+):', self.message)
                        if fail_m:
                            self.event_type = fail_m.group(1)
                            self.interface = fail_m.group(2)
                            
                ip_m = re.search(r'(?:from|to|bound to)\s+([0-9]{1,3}(?:\.[0-9]{1,3}){3})', self.message)
                if ip_m:
                    self.ip = ip_m.group(1)

                    elif self.message.startswith('bound to'):
