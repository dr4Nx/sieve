    with open_fn(file_path, 'rt', encoding='utf-8', errors='replace') as f:
        for line in f:
            line = line.strip()
            m = log_pattern.match(line)
            if not m:
                continue
            
            host = m.group(1)
            # If the extracted host is actually a time string, default to localhost (syslog missing host field)
            if re.match(r"^\d{2}:\d{2}:\d{2}(?:\.\d+)?$", host):
                host = "localhost"
                
            msg = m.group(2)
            
            sm = success_pattern.search(msg)
            if sm:
                stats[host]['success'].add(sm.group(1))
                continue
                
            fm = failed_pattern.search(msg)
            if fm:
                # Find the first non-None captured group corresponding to the IP
                ip = next(g for g in fm.groups() if g is not None)
                stats[host]['failed'].add(ip)

    for host, data in stats.items():
        print(f"{host} {len(data['failed'])} {len(data['success'])}")
