        r'Address (?P<ip>\S+) maps to',
        r'\[(?P<ip>[0-9a-fA-F\.\:-]+)\] failed'

def extract_ip_from_payload(payload):

    if m:

    if m:
        return m.group(1)

    m = re.search(r'Address\s+([^\s]+)\s+maps to', payload)
    if m:
        return m.group(1)

    for m in re.finditer(r'(?:from|by)\s+([^\s]+)', payload):
        cand = m.group(1)
        if cand.endswith(':') and not cand.endswith('::'):
