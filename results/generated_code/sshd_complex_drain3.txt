def process_file(filepath):

    syslog_re = re.compile(r'\d{2}:\d{2}:\d{2}[^\s]*\s+(?P<host>\S+)\s+sshd(?:-session)?(?:\[\d+\]|\([^)]+\)(?:\[\d+\])?)?:\s*(?P<msg>.*)')
