def extract_ip(line):
    m = re.search(r'rhost=([^\s]+)', line)
    if m:
        ip = m.group(1).rstrip(':')
        if ip.lower() not in excluded_ips:
            return ip
    
    for m in re.finditer(r'from ([^\s]+)', line):
        ip = m.group(1).rstrip(':')
        if ip.lower() not in excluded_ips:
            return ip

    for m in re.finditer(r'by ([^\s]+)', line):
        ip = m.group(1).rstrip(':')
        if ip.lower() not in excluded_ips:
            return ip
            
    return None

FROM_RE = re.compile(r'from\s+([a-zA-Z0-9.\-:]+)')
ON_RE = re.compile(r'on\s+([a-zA-Z0-9.\-]+)\.$')

FAILED_RE = re.compile(r'^Failed (?!none\b)\S+ for ')
INVALID_USER_RE = re.compile(r'^Invalid user ')
ACCEPTED_RE = re.compile(r'^Accepted \S+ for ')
SESSION_OPENED_RE = re.compile(r'^session opened for user ')
PID_RE = re.compile(r'^(?:\([^)]+\))?\[(\d+)\]:\s*(.*)')
NO_PID_RE = re.compile(r'^:\s*(.*)')

def extract_ip(msg):
    m = IPV4_RE.search(msg)
    if m: return m.group(0)
    
    m = IPV6_RE.search(msg)
    if m: return m.group(0)
    
    m = RHOST_RE.search(msg)
    if m: return m.group(1)
    
    m = FROM_PORT_RE.search(msg)
    if m: return m.group(1)
    
    m = CLOSED_BY_RE.search(msg)
    if m:
        val = m.group(1)
        if val.lower() not in ('remote', 'peer', 'user', 'port', 'invalid', 'authenticating'):
            return val
            
    m = FROM_RE.search(msg)
    if m:
        val = m.group(1)
        if val.lower() not in ('remote', 'peer', 'user', 'port', 'invalid', 'authenticating', 'socket', 'system'):
            if '.' in val or ':' in val or val.isalnum():
                return val
                
    if 'Rhosts with RSA host authentication accepted' in msg:
        m = ON_RE.search(msg)
        if m: return m.group(1)
        
    return None

def main():
    if len(sys.argv) < 2:
        return
        
    file_path = sys.argv[1]
    
    host_stats = defaultdict(lambda: {"success_ips": set(), "fail_ips": set()})
    pid_ip_map = {}
    
    open_func = gzip.open if file_path.endswith('.gz') else open
        
    with open_func(file_path, 'rt', encoding='utf-8', errors='replace') as f:
        for line in f:
            idx = line.find(' sshd')
            if idx == -1:
                continue
                
            left_part = line[:idx].rstrip()
            host_space_idx = left_part.rfind(' ')
            if host_space_idx == -1:
                host = left_part
            else:
                host = left_part[host_space_idx+1:]
                
            right_part = line[idx+5:] 
            pid = None
            msg = right_part
            
            m_pid = PID_RE.match(right_part)
            if m_pid:
                pid = m_pid.group(1)
                msg = m_pid.group(2)
            else:
                m_pid2 = NO_PID_RE.match(right_part)
                if m_pid2:
                    msg = m_pid2.group(1)
            
            ip = extract_ip(msg)
            if ip and pid:
                pid_ip_map[(host, pid)] = ip
                
            if not ip and pid:
                ip = pid_ip_map.get((host, pid))
                
            is_failure = (
                FAILED_RE.search(msg) or
                INVALID_USER_RE.search(msg) or
                'authentication failure;' in msg or
                'more authentication failure' in msg or
                'error: PAM: Authentication failure' in msg or
                'bad username [' in msg or
                'check pass; user unknown' in msg
            )
            
            is_success = (
                ACCEPTED_RE.search(msg) or
                'authentication success;' in msg or
                'authentication succeeds' in msg or
                SESSION_OPENED_RE.search(msg) or
                'pam_unix(sshd:session): session opened for user' in msg or
                'Rhosts with RSA host authentication accepted' in msg
            )
            
            if is_failure and ip:
                host_stats[host]['fail_ips'].add(ip)
            if is_success and ip:
                host_stats[host]['success_ips'].add(ip)
                
    for host in sorted(host_stats.keys()):
        stats = host_stats[host]
        print(f"{host} {len(stats['fail_ips'])} {len(stats['success_ips'])}")

if __name__ == '__main__':
    main()
