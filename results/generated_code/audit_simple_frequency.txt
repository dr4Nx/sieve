            for line in f:
                # Filter for service start events (type 1130 is SERVICE_START)
                if not re.search(r'\btype=(?:1130|SERVICE_START)\b', line):

                # Extract the unit name. It might be explicitly given in unit=
                unit_match = re.search(r'\bunit=(?:"([^"]+)"|([^\s\'"]+))', line)

                        if comm_match:
                            comm = comm_match.group(1)
                            # 'systemd' itself isn't the unit name being started
                            if comm != "systemd":
                                print(comm)
    except Exception:
        pass
