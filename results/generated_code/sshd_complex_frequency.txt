def to_seconds(month, day, H, M, S):
    # Using year 2000 (leap year) per instructions
    days_in_month = [0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    day_of_year = sum(days_in_month[:month]) + day - 1
    return day_of_year * 86400 + H * 3600 + M * 60 + S
