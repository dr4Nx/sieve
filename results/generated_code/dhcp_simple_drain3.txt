    # and specific log messages like 'dhclient(PID) is already running'.
    pid_regex = re.compile(r'\b\w+\[(\d+)\]:|dhclient\((\d+)\)')

            for line in f:

                    pid_str = match.group(1) or match.group(2)
                    if pid_str and int(pid_str) > 5000:
                        sys.stdout.write(line)
                        break
    except Exception as e:
        sys.stderr.write(f"Error processing {file_path}: {e}\n")
