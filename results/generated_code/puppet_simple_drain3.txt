def filter_puppet_reenabled(log_path):
    # Regex to parse the standard syslog fields
    syslog_re = re.compile(r'^([A-Z][a-z]{2}\s+\d+\s+\d{2}:\d{2}:\d{2})\s+(\S+)\s+([^:]+):\s+(.*)$')
    
    # Dictionary to keep track of puppet agent state per host
    # host -> boolean (True if disabled)
    puppet_disabled = {}
    
    try:
        with open(log_path, 'r', encoding='utf-8', errors='ignore') as f:
            for line in f:
                line_str = line.strip()
                match = syslog_re.match(line_str)
                if not match:
                    continue
                    
                record = {
                    'timestamp': match.group(1),
                    'host': match.group(2),
                    'process': match.group(3),
                    'msg': match.group(4),
                    'line': line_str
                }
                
                # Check for disable events
                if "Disabling Puppet." in record['msg'] or "administratively disabled" in record['msg']:
                    puppet_disabled[record['host']] = True
                
                # Check for enable events
                elif "Enabling Puppet." in record['msg']:
                    # The message "Enabling Puppet." is the indicator that execution was re-enabled.
                    # We output the line matching this event.
                    print(record['line'])
                    puppet_disabled[record['host']] = False
    except IOError:
        pass
