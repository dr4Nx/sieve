    return datetime.strptime(t_str, "%Y %b %d %H:%M:%S").timestamp()

def main():
    if len(sys.argv) < 2:
        return
    
    line_re = re.compile(r'^(?:<\d+>)?([A-Z][a-z]{2})\s+(\d+)\s+(\d{2}:\d{2}:\d{2})\s+(\S+)\s+(.*)')
    base_re = re.compile(r'^([a-zA-Z0-9_-]+)(?:\[(\d+)\])?:\s+(.*)')
    nested_re = re.compile(r'^([A-Z][a-z]{2})\s+(\d+)\s+(\d{2}:\d{2}:\d{2})\s+([a-zA-Z0-9_-]+)(?:\[(\d+)\])?:\s+(.*)')
    bound_re = re.compile(r'bound to ([0-9a-fA-F:\.]+)\s+--\s+renewal in (\d+)\s+seconds')
    
    state = {}
    
    open_func = gzip.open if sys.argv[1].endswith('.gz') else open
    with open_func(sys.argv[1], 'rt', encoding='utf-8', errors='ignore') as f:
        for line in f:
            line = line.strip()
            m = line_re.match(line)
            if not m:
                continue
                
            base_month, base_day, base_time, host, rest = m.groups()
            base_ts = parse_time(base_month, base_day, base_time)
            
            parts = re.split(r'<\d+>', rest)
            
            for i, part in enumerate(parts):
                part = part.strip()
                if not part:
                    continue
                    
                if i == 0:
                    m_base = base_re.match(part)
                    if m_base:
                        proc_name, pid, msg = m_base.groups()
                        process_msg(host, proc_name, pid, base_ts, msg, state, bound_re)
                else:
                    m_nested = nested_re.match(part)
                    if m_nested:
                        n_month, n_day, n_time, proc_name, pid, msg = m_nested.groups()
                        ts = parse_time(n_month, n_day, n_time)
                        process_msg(host, proc_name, pid, ts, msg, state, bound_re)

def process_msg(host, proc_name, pid, ts, msg, state, bound_re):
    pid = pid if pid else ""
    key = (host, proc_name, pid)
    
    b_m = bound_re.match(msg)
    if b_m:
        assigned_ip = b_m.group(1)
        renewal_time = float(b_m.group(2))
        state[key] = (ts, assigned_ip, renewal_time)
        return
        
    if msg.startswith('DHCPREQUEST'):
        if key in state:
            bound_ts, assigned_ip, renewal_time = state[key]
            elapsed = ts - bound_ts
            if 0 <= elapsed <= 0.25 * renewal_time:
                print(f"{host} {assigned_ip} {float(elapsed)}")
                del state[key]

if __name__ == '__main__':
    main()
