def extract_ip_for_state(msg):
    m = re.search(r'\brhost=([^\s]+)', msg)
    if m:
        return m.group(1)
        
    m = re.search(r'(?:Connection from|from|Connection closed by|Connection reset by|Disconnect(?:ed)? from)\s+([^\s]+)', msg)
    if m:
        val = m.group(1).rstrip(':,]')
        if val.lower() not in ('remote', 'user', 'the', '(uid=0)', 'port') and '_u:' not in val:
            if not re.match(r'^[\W_]+$', val):
                return val
                
    m = re.search(r'Address ([^\s]+) maps to', msg)
    if m:
        return m.group(1)
        
    m = re.search(r'\[([^\s]+)\] failed - POSSIBLE BREAK-IN ATTEMPT', msg)
    if m:
        return m.group(1)

    return None

def is_auth_failure(msg):
    msg_l = msg.lower()
    if 'authentication failure' in msg_l:
        return True
    if msg.startswith('Failed '):
        return True
    if 'invalid user' in msg_l:
        return True
    if 'bad username' in msg_l:
        return True
    if 'user unknown' in msg_l:
        return True
    return False
