def get_ts(mon, day, hr, mn, sc):
    month = get_month(mon)
    day = int(day)
    try:
        dt = datetime.datetime(2000, month, day, int(hr), int(mn), int(sc))
    except ValueError:
        if month == 2 and day == 29:
            dt = datetime.datetime(2000, 2, 29, int(hr), int(mn), int(sc))
        else:
            dt = datetime.datetime(2000, month, min(day, 28), int(hr), int(mn), int(sc))
    return int(dt.timestamp())

def get_peak_failures_fast(events, window_sec):
    events.sort(key=lambda x: x[0])
    max_count = 0
    current_count = 0
    left = 0
    
    for right in range(len(events)):
        current_count += events[right][1]
        
        while events[right][0] - events[left][0] > window_sec:
            current_count -= events[left][1]
            left += 1
            
        if current_count > max_count:
            max_count = current_count
            
    return max_count

import gzip
import re
from collections import defaultdict

def main():
    if len(sys.argv) < 2:
        return
    filepath = sys.argv[1]
    
    # Dictionary to store distinct IPs for success and failure per host
    data = defaultdict(lambda: {"success": set(), "fail": set()})
    
    # Regex to parse the base syslog pattern: timestamp, host, and sshd message
    line_re = re.compile(r'^.*?\s+(?P<host>\S+)\s+sshd(?:\[\d+\]|\([^)]+\)\[\d+\])?:\s+(?P<msg>.*)')
    
    # Regexes for successful authentications (capturing source IP)
    succ_accepted_re = re.compile(r'Accepted \S+ for .*?from (\S+)')
    succ_pam_re = re.compile(r'authentication success;.*?\brhost=(\S+)')
    
    # Regexes for failed authentications (capturing source IP)
    fail_failed_re = re.compile(r'Failed \S+ for .*?from (\S+)')
    fail_pam_re = re.compile(r'authentication failure.*?\brhost=(\S+)')
    fail_pam_err_re = re.compile(r'Authentication failure for .*?from (\S+)')
    fail_invalid_re = re.compile(r'Invalid user .*?from (\S+)')
    fail_not_allowed_re = re.compile(r'User .*?from (\S+) not allowed')

    open_func = gzip.open if filepath.endswith('.gz') else open

    try:
        with open_func(filepath, 'rt', errors='ignore') as f:
            for line in f:
                m = line_re.match(line)
                if not m:
                    continue
                
                host = m.group('host')
                msg = m.group('msg')
                
                # Check for successes
                m_succ = succ_accepted_re.search(msg)
                if m_succ:
                    data[host]["success"].add(m_succ.group(1))
                    continue
                
                m_succ = succ_pam_re.search(msg)
                if m_succ:
                    data[host]["success"].add(m_succ.group(1))
                    continue
                    
                # Check for failures
                m_fail = fail_failed_re.search(msg)
                if m_fail:
                    data[host]["fail"].add(m_fail.group(1))
                    continue
                    
                m_fail = fail_pam_re.search(msg)
                if m_fail:
                    data[host]["fail"].add(m_fail.group(1))
                    continue
                    
                m_fail = fail_pam_err_re.search(msg)
                if m_fail:
                    data[host]["fail"].add(m_fail.group(1))
                    continue
                    
                m_fail = fail_invalid_re.search(msg)
                if m_fail:
                    data[host]["fail"].add(m_fail.group(1))
                    continue
                    
                m_fail = fail_not_allowed_re.search(msg)
                if m_fail:
                    data[host]["fail"].add(m_fail.group(1))
                    continue
    except Exception:
        pass

    # Output the results for each host
    for host in data:
        f_count = len(data[host]["fail"])
        s_count = len(data[host]["success"])
        if f_count > 0 or s_count > 0:
            print(f"{host} {f_count} {s_count}")
