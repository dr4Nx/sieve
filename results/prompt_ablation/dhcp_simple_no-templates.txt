    # Therefore, a lease duration > 1 day (86400s) typically corresponds to a renewal time > 12 hours (43200s).

    ips = set()

            for line in f:
                m = bound_re.search(line)
                if m:
                    ip = m.group(1)
                    renewal_secs = int(m.group(2))
