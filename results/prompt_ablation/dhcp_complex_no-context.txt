def parse_time(ts_str):
    ts_str_norm = re.sub(r'\s+', ' ', ts_str.strip())
    try:
        if 'T' in ts_str_norm:
            ts_str_norm = ts_str_norm.replace('Z', '+00:00')
            return datetime.fromisoformat(ts_str_norm).timestamp()
        elif re.search(r'[a-zA-Z]', ts_str_norm):
            if ts_str_norm[0].isdigit():
                return datetime.strptime(ts_str_norm, "%Y %b %d %H:%M:%S").timestamp()
            else:
                return datetime.strptime(f"2000 {ts_str_norm}", "%Y %b %d %H:%M:%S").timestamp()
        else:
            ts_str_iso = ts_str_norm.replace(' ', 'T')
            return datetime.fromisoformat(ts_str_iso).timestamp()
    except ValueError:
        pass
    return None

    months = {'Jan':1, 'Feb':2, 'Mar':3, 'Apr':4, 'May':5, 'Jun':6, 'Jul':7, 'Aug':8, 'Sep':9, 'Oct':10, 'Nov':11, 'Dec':12}

def parse_time(ts_str):
    parts = ts_str.split()
    month = parse_month(parts[0])
    day = int(parts[1])
    h, m, s = map(int, parts[2].split(':'))
    # Timeline starts in 2000 per instructions; avoiding year rollover complexities
    dt = datetime.datetime(2000, month, day, h, m, s)
    return dt.timestamp()
