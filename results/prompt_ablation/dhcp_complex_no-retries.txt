MONTHS = {"Jan": 1, "Feb": 2, "Mar": 3, "Apr": 4, "May": 5, "Jun": 6,

def get_timestamp(month_str, day_str, time_str):
    month = MONTHS[month_str[:3]]
    day = int(day_str)
    H, M, S = map(int, time_str.split(':'))
    dt = datetime.datetime(2000, month, day, H, M, S)
    return int(dt.timestamp())

def process_file(filepath):

        r'(?P<day>\d+)\s+'
        r'(?P<time>\d{2}:\d{2}:\d{2})\s+'

        r'dhclient(?:\[(?P<pid>\d+)\])?:\s+'

    )

                 'Jul': 7, 'Aug': 8, 'Sep': 9, 'Oct': 10, 'Nov': 11, 'Dec': 12}

    # Track latest bound state per host/process
    # key: (host, pid), value: {'bound_ts': float, 'ip': str, 'ren': float}
    state = {}

    opener = gzip.open if filepath.endswith('.gz') else open
    with opener(filepath, 'rt', encoding='utf-8', errors='ignore') as f:
        for line in f:
            line = line.strip()

                continue
            
            m = log_pattern.match(line)

                continue

            day = m.group('day')
            time_str = m.group('time')

            pid = m.group('pid')
            msg = m.group('msg')

                d = int(day)
