        return dt.timetuple().tm_yday * 86400 + dt.hour * 3600 + dt.minute * 60 + dt.second
    except ValueError:
        return -1

def parse_duration(val_str, unit_str):
    val = int(val_str)
    if not unit_str:
        return val
    u = unit_str.lower()
    if u.startswith('s'): return val
    if u.startswith('mo'): return val * 2592000
    if u.startswith('m'): return val * 60
    if u.startswith('h'): return val * 3600
    if u.startswith('d'): return val * 86400
    if u.startswith('w'): return val * 604800
    if u.startswith('y'): return val * 31536000
    return val

def process_log(file_path):
    # Pattern to detect lines that report an IP assignment
    assignment_pattern = re.compile(r'\b(DHCPACK|bound\s+to|leased|lease\s+of)\b', re.IGNORECASE)
    
    unit_regex = r'(second|sec|s|minute|min|m|hour|hr|h|day|d|week|w|month|mo|year|y)s?'
    
    # Explicit lease duration mentions like "lease time 86405", "for 2 days"
    explicit_pattern = re.compile(
        r'\b(?:lease\s+time|lease\s+duration|lease|duration|for)\s+(?:is\s+|:\s+)?(\d+)(?:\s*' + unit_regex + r'\b)?',
        re.IGNORECASE
    )
    
    # Infinite lease explicitly
    infinite_pattern = re.compile(
        r'\b(?:lease\s+time|lease\s+duration|lease|duration|for)\s+(?:is\s+|:\s+)?(infinite|infinity|forever)\b',
        re.IGNORECASE
    )

    # Renewal time which is typically 50% of the lease duration
    renewal_pattern = re.compile(
        r'\brenewal\s+in\s+(\d+)(?:\s*' + unit_regex + r'\b)?',
        re.IGNORECASE
    )

    with open(file_path, 'r', encoding='utf-8', errors='replace') as f:
        for line in f:
            # Only evaluate lines that report an IP assignment
            if not assignment_pattern.search(line):
                continue

            found = False
            
            if infinite_pattern.search(line):
                found = True
            
            if not found:
                for m in explicit_pattern.finditer(line):
                    secs = parse_duration(m.group(1), m.group(2))
                    if secs > 86400:
                        found = True
                        break

                    # Since renewal is typically half the lease time, > 43200 implies lease > 1 day
                    if secs > 43200:
                        found = True
                        break

            if found:

    msg_regex = re.compile(r'\b(DHCP(?:DISCOVER|OFFER|REQUEST|DECLINE|ACK|NAK|RELEASE|INFORM))\b(?:\s*\(([^)]+)\))?')
    
    # Match potential interfaces following 'on' or 'via' (e.g., ISC DHCP format)
    on_via_regex = re.compile(r'\b(?:on|via)\s+([a-zA-Z0-9_.:-]+)\b')
    
    # Exclude patterns to filter out IPs and MACs mistakenly captured as interfaces
    ipv4_regex = re.compile(r'^\d{1,3}(?:\.\d{1,3}){3}$')
    hex_colon_regex = re.compile(r'^[0-9a-fA-F:]+$')

    open_func = gzip.open if filepath.endswith('.gz') else open

    try:
        with open_func(filepath, 'rt', encoding='utf-8', errors='ignore') as f:
            for line in f:
                m = msg_regex.search(line)
                if not m:
                    continue
                
                msg_type = m.group(1)
                raw_interface = m.group(2)
                
                interface = None
                # Validate parenthesized interface if present
                if raw_interface:
                    if not ('=' in raw_interface or ipv4_regex.match(raw_interface) or (':' in raw_interface and hex_colon_regex.match(raw_interface))):
