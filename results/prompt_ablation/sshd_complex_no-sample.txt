def is_ip(s):
    if re.match(r'^(\d{1,3}\.){3}\d{1,3}$', s):
        return True
    if ':' in s and re.match(r'^[0-9a-fA-F:%]+$', s):
        return True
    return False

def extract_ip_from_msg(msg):
    m = re.search(r'from (\S+)', msg)

    m = re.search(r'by (\S+)', msg)
    if m and is_ip(m.group(1)):

    return None
