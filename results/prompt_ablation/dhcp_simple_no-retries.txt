    )

    with open(filepath, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            for match in pattern.finditer(line):
                ip_str = match.group(1) or match.group(2) or match.group(3)
                if not ip_str:
                    continue

                ip_str = ip_str.rstrip('.')
                
                try:

                        continue
                    # Also exclude multicast addresses (e.g., IPv6 ff00::/8)
                    if ip_obj.is_multicast:
                        continue
                    
                    print(ip_str)
                except ValueError:
                    # Not a valid IP address
                    pass
