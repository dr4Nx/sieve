            except ValueError:
                pass

            if ts_str.endswith('Z'):
                ts_str_clean = ts_str[:-1] + '+0000'

def process_logs(filepath):
    execution_counts = defaultdict(int)
    
    open_func = gzip.open if filepath.endswith('.gz') else open
    with open_func(filepath, 'rt', encoding='utf-8', errors='ignore') as f:
        for line in f:
            # Identify command executions by matching the word CMD
            if not re.search(r'\bCMD\b', line):
                continue
                
            # Match ISO 8601-like timestamps
            m_iso = re.match(r'^(\d{4}-\d{2}-\d{2})[T\s](\d{2}):(\d{2}):\d{2}', line)
            # Match standard syslog timestamps
            m_syslog = re.match(r'^([A-Z][a-z]{2}\s+\d+)\s+(\d{2}):(\d{2}):\d{2}', line)
            
            date_val = None
            hour = None
            minute = None
            
            if m_iso:
                date_val = m_iso.group(1)
                hour = int(m_iso.group(2))
                minute = int(m_iso.group(3))
            elif m_syslog:
                date_val = re.sub(r'\s+', ' ', m_syslog.group(1))
                hour = int(m_syslog.group(2))
                minute = int(m_syslog.group(3))
                
            if date_val is not None and hour is not None and minute is not None:
                interval_minutes = hour * 60 + (minute // 15) * 15
                execution_counts[(date_val, interval_minutes)] += 1
                
    for (date_val, interval_minutes) in sorted(execution_counts.keys()):
        print(f"{date_val} {interval_minutes} {execution_counts[(date_val, interval_minutes)]}")
