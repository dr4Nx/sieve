def process_cron_logs(file_path):

    # Regex matches ISO and Syslog format timestamps, then looks for the execution marker 'CMD ('
    log_pattern = re.compile(
        r'^(?:(\d{4}-\d{2}-\d{2})[T\s]|([A-Z][a-z]{2}\s+\d+)\s+)(\d{2}):(\d{2}):\d{2}.*?\bCMD\s*\('
    )

    try:
        with open_func(file_path, 'rt', encoding='utf-8', errors='ignore') as f:
            for line in f:
                m = log_pattern.search(line)
                if m:
                    date1, date2, hour_str, min_str = m.groups()
                    date = date1 if date1 else date2
                    hour = int(hour_str)
                    minute = int(min_str)
                    
                    # Calculate 15-minute interval as minutes since midnight
                    interval = hour * 60 + (minute // 15) * 15
                    counts[(date, interval)] += 1
    except Exception:
        pass

    # Output space-separated results: date, interval_minutes, execution_count
    for (date, interval) in sorted(counts.keys()):
        print(f"{date} {interval} {counts[(date, interval)]}")
