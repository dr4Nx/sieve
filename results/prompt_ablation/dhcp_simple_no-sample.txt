def process_file(filepath):
    # Regex to match the standard syslog prefix for dhclient
    # Examples:
    # May 11 11:15:26 localhost dhclient: ...
    # Mar 10 00:01:48 puma25 dhclient[11386]: ...
    prefix_re = re.compile(r'^[A-Z][a-z]{2}\s+\d+\s+(?:\d{4}\s+)?\d{2}:\d{2}:\d{2}\s+(\S+)\s+dhclient(?:\[(\d+)\])?:\s+(.*)')
    
    ips = set()
    # Map to track the IP being tried for a specific host and process ID
    state = {} # (host, pid) -> IP
    
    try:
        with open(filepath, 'r', encoding='utf-8', errors='replace') as f:
            for line in f:
                m = prefix_re.search(line)
                if m:
                    host = m.group(1)
                    pid = m.group(2)
                    msg = m.group(3)
                else:
                    host = None
                    pid = None
                    msg = line
                
                key = (host, pid)
                
                # Track "Trying recorded lease <IP>" to infer IPs for subsequent lines that lack them
                if 'Trying recorded lease' in msg:
                    tm = re.search(r'Trying recorded lease\s+([0-9a-fA-F\.:]+)', msg)
                    if tm:
                        state[key] = tm.group(1)
                
                # Check for "renewal in X seconds" which represents the lease duration status
                if 'renewal in ' in msg:
                    ren_m = re.search(r'renewal in (\d+) seconds', msg)
                    if ren_m:
                        duration = int(ren_m.group(1))
                        ip = None
                        
                        # The IP is usually stated right before "renewal in"
                        ip_m = re.search(r'bound to ([0-9a-fA-F\.:]+)', msg)
                        if ip_m:
                            ip = ip_m.group(1)
                        else:
                            # Fallback to the last IP we saw being tried for this host/pid
                            ip = state.get(key)
                            
                        # "over a day" strictly means > 86400 seconds
                        if ip and duration > 86400:
                            ips.add(ip)
                        
                        # Update state for any potential interleaved logs
                        if ip:
                            state[key] = ip
                            
    except Exception as e:
        pass
        
    for ip in sorted(list(ips)):
        print(ip)
