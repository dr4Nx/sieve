def process_logs():
    if len(sys.argv) < 2:
        return
    file_path = sys.argv[1]
    
    # Robust syslog extraction for host and sshd message body
    # Handles varying timestamp formats natively by scanning until the hostname and 'sshd'
    log_line_re = re.compile(r'^.*?\s+(\S+)\s+sshd(?:\[\d+\]|\([^)]+\)(?:\[\d+\])?)?:\s+(.*)')
    
    failed_patterns = [
        re.compile(r'authentication failure;.* rhost=([^\s]+)'),
        re.compile(r'Failed \S+ for (?:invalid user )?\S+ from ([^\s]+)'),
        re.compile(r'error: PAM: Authentication failure for \S+ from ([^\s]+)'),
        re.compile(r'Invalid user \S+ from ([^\s]+)'),
        re.compile(r'\d+ more authentication failure(?:s)?;.* rhost=([^\s]+)'),
        re.compile(r'User \S+ from ([^\s]+) not allowed because')
    ]

    success_patterns = [
        re.compile(r'Accepted \S+ for \S+ from ([^\s]+)'),
        re.compile(r'authentication success;.* rhost=([^\s]+)')
    ]
    
    hosts_data = collections.defaultdict(lambda: {'failed': set(), 'success': set()})
    
    open_func = gzip.open if file_path.endswith('.gz') else open
    
    try:
        with open_func(file_path, 'rt', encoding='utf-8', errors='replace') as f:
            for line in f:
                match = log_line_re.search(line)
                if not match:
                    continue
                    
                host = match.group(1)
                msg = match.group(2)
                
                matched = False
                for p in success_patterns:
                    m = p.search(msg)
                    if m:
                        ip = m.group(1)
                        hosts_data[host]['success'].add(ip)
                        matched = True
                        break
                        
                if not matched:
                    for p in failed_patterns:
                        m = p.search(msg)
                        if m:
                            ip = m.group(1)
                            hosts_data[host]['failed'].add(ip)
                            break
                            
    except Exception:
        pass
        
    for host in sorted(hosts_data.keys()):
        f_count = len(hosts_data[host]['failed'])
        s_count = len(hosts_data[host]['success'])
